CREATE TABLE users (
  id INT(11) AUTO_INCREMENT PRIMARY KEY,
  first_name VARCHAR(50),
  last_name VARCHAR(50),
  number VARCHAR(20),
  email VARCHAR(100),
  emergency VARCHAR(20),
  username VARCHAR(50),
  password VARCHAR(255),
  month VARCHAR(20),
  day INT(2),
  year INT(4),
  gender VARCHAR(10),
  terms_accepted TINYINT(1) DEFAULT 0
);

CREATE TABLE appointments (
  id INT(11) AUTO_INCREMENT PRIMARY KEY,
  user_id INT(11),
  first_name VARCHAR(50),
  last_name VARCHAR(50),
  email VARCHAR(100),
  service_name VARCHAR(100),
  appointment_date DATE,
  appointment_time TIME,
  status VARCHAR(20) DEFAULT 'Pending',
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE missed_appointments (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    user_id INT(11),
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    email VARCHAR(100),
    service_name VARCHAR(100),
    appointment_date DATE,
    appointment_time TIME,
    status VARCHAR(20) DEFAULT 'Missed',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

INSERT INTO missed_appointments (user_id, name, email, service_name, appointment_date, appointment_time)
SELECT user_id, name, email, service_name, appointment_date, appointment_time
FROM appointments
WHERE status = 'Pending' AND appointment_date < CURDATE();
